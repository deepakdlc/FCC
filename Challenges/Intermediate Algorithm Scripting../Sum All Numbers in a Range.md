function sumAll(arr) {
  let sum = 0;
  let min = arr[0];
  let max = arr[1];
  
  if(min > max){  // Swapping
    [min,max] = [max,min]
  }

  for(let i = min; i <= max; i++){
    sum += i;    
  }

  return sum;
}

sumAll([1, 4]);
